import random


print("==========第一题，不重复随机数==========")
n = eval(input("请输入正整数n:"))
lst1 = random.sample(range(1, 5 * n + 1), n)
print("产生随机数为：", lst1)
lst2 = [i for i in lst1 if i % 2 == 0]
print("其中偶数有：", lst2)

print("==========第二题，切片排序==============")
lst = [random.randint(0, 20) for i in range(20)]
print(lst)
a = lst[0:10]
a.sort()
b = lst[10:]
b.sort(reverse=True)
lst = a + b
print(lst)
